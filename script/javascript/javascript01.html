01. 변수 : 데이터 저장

{
    var x = 100;          //변수 x에 숫자 100을 저장
    var y = 200;          //변수 y에 숫자 200을 저장
    var z = "javascript"; //변수 z에 문자 javascript를 저장

    document.write(** 01. 변수 **);
    document.write(x);
    document.write(y);
    document.write(z);
}

02. 변수 : 데이터 저장 + 데이터 변경

{
    let x = 100;
    let y = 200;
    let z = "javascript"
    x = 300;   // 아래 나중에 입력한 값으로 데이터 변경(변경시에는 let이나 var 쓰지않음 쓸경우 에러남)
    y = 400;
    z = "jquery";

    document.write("*** 02. 변수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
}

03. 변수 : 데이터 저장 + 데이터 변경 + 데이터 추가

{
    let x = 100;
    let y = 200;
    let z = "javascript";

    x += 300; //기존의 저장값에 데이터를 추가 해줄수있음
    y += 400;
    z += "jquery"; //문자의 경우 숫자와 달리 더하는게 불가능하므로 뒤에 문자열이 추가됨

    document.write("*** 03. 변수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
}

04. 변수 : 변수의 종류 = 지역변수 + 전역변수 + 매개변수

{
    let x = 100;
    let y = 200;
    let z = "javascript";
    
    document.write("
*** 04. 변수: 변수의종류 ***
");
    document.write(x);
    document.write(y);
    document.write(z);
}

05. 상수 : 데이터 저장소 + 데이터 변경이 불가함

{
    const x = 100;
    const y = 200;
    const z = "javascript";
    
    //x= 300;
    //y= 400;
    //z= "Jquery";
    
    document.write("
*** 05. 상수 ***
");
    document.write(x,);
    document.write(y,);
    document.write(z,);
}

06. 배열 : 여러개의 데이터를 저장 : 표현방법1

{
    const arr = new Array();
    arr[0] = 100;
    arr[1] = 200;
    arr[2] = "javascript";
    
    document.write("
*** 06. 배열 ***
");
    document.write(arr[0],);
    document.write(arr[1],);
    document.write(arr[2],);
}

07. 배열 : 여러개의 데이터를 저장 : 표현방법2

{
    const arr = new Array(100, 200, "javascript"); // new Array()괄호안에 값을 넣음으로써 arr[0] 쓰는것을 생략할수있음

    document.write("
*** 07. 배열 ***
");
    document.write(arr[0],);
    document.write(arr[1],);
    document.write(arr[2],);
}
