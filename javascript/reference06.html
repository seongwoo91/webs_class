조건문

조건문은 조건식의 값이 참(true)인지, 거짓(false)인지에 따라 JS코드를 제어합니다.
예를 들어 블로그에서 방문자를 대상으로 좋아하는 숫자를 입력받습니다. 이때 입력한 값을 2로 나누어 나머지 값이 0이면 "좋아하는 수는 짝수" 라고 출력하게 되고,
아닐 경우에는 "좋아하는 수는 홀수" 라고 출력하게 하려고 할때, 이렇게 수를 나눈 나머지 값에 따라 다른 결과를 출력하게 하고 싶다면 조건문을 사용해야 합니다.
조건문의 종류에는 if문, else문, if else문이 있으며 중첩하여 사용할수도 있습니다.

if문

if문은 조건을 만족할 경우에만 코드를 실행합니다.
조건식은 어떤 데이터를 입력해도 true 또는 false를 반환합니다.

if(조건식){
    자바스크립트 코드;
}

적용 예제

var num=10;
if(num<100){       //true를 반환합니다.
    document.write("hello");
}
if(0){                //false를 반환합니다.
    document.write("hello")
}

다음 예제는 방문자로부터 입력받은 수를 조건문으로 만들고,하루 운동시간이 30분 이상일 경우에만 결과를 출력하도록 작성했습니다.

<script>
    var exemin = prompt("하루에 몇분 정도 운동하시나요?", "0");

    if(exemin >= 30) {
        document.write("건강관리를 잘하고 계시네요!". "<br>");
    }
    document.write("== The End ==")
<script>

위의 예제에서 if(exemin >= 30) { -- }는 질의응답창에 30 이상 입력했을 경우에 중괄호 ({ -- })의 코드를 실행합니다.
하지만 30 이하일 경우에는 조건이 맞지 않기 때문에'== The End =='만 화면에 출력합니다.



조건식에 논리형 데이터가 아닌 다른 형이 오는 경우

조건식에 논리형 데이터(true, false)가 아닌 다른 형의 데이터가 입력되어도 true 또는 false로 인식됩니다.
다음은 조건식에 논리형 데이터가 아닌 다른 형의 데이터가 입력되었을때 반환되는 결과입니다.


다음 값이 조건식에 입력되면 false를 반환하지만 그 밖의 값은 true로 인식합니다.

0, null, ""(빈 문자), undifined

출력되는경우

var num=3;
if(num){    //3은 true입니다.
    document.write(num);
}

출력되지 않는 경우

var num=0;
if(num){    //0은 false입니다.
    document.write(num);
}

다음 예제에서 방문자가 질의응답 창에 이름을 입력하지 않으면 기본 문자("")가 변수에 저장되어 조건식에 거짓(false)값이 반환되고 그 결과 문장이 출력되지 않습니다.

<script>
    var user_name = prompt("방문자의 이름은?","");

    if(user_name){    //user_name에 이름이 저장되면 중괄호의 코드가 실행됩니다.
        document.write(user_name+"님 반갑습니다! 방문을 환영합니다.")
    }
<script>


else 문

else 문은 조건식을 만족할(true) 경우와 만족하지 않을(false)경우에 따라 실행되는 코드가 달라집니다.
즉, 두가지 결과가 나올수 있습니다.
아래는 조건식의 만족 여부에 따라 실행되는 코드가 달라지는 else 문의 기본형입니다.

if(조건식){
    자바스크립트 코드1;    
} else {
    자바스크립트 코드2;
}

다음 예제를 통해 else문에 대해 알아보겠습니다. 방문자에게 숫자를 받아서 if~ else 문을 사용하여 입력된 값이 짝수,홀수 인지에 따라 출력되는 결과가 다르도록 하였습니다.

<script>
    var num = prompt("해당 숫자는?", "0");

    if(num % 2 == 0) {  //짝수일 경우에 실행
        documnet.write("짝수입니다.");
    } else {            //홀수일 경우에 실행
        document.write("홀수입니다.");
    }
</script>

다음은 confirm 객체를 사용하여 else 조건문을 더 알아보겠습니다. conrirm 객체를 사용하여 웹페이지에 회원 탈퇴 여부를 묻는 확인/취소 창을 띄우게 하고,
else 조건문으로 사용자가 [확인] 버튼을 눌렀을때와 [취소] 버튼을 눌렀을 때 결과 화면이 다르게 나타나도록 할 것입니다.

<script>
    var result = confirm("정말로 회원을 탈퇴하시겠습니까?");

    if(result) {
        document.write("탈퇴 처리되었습니다!");
    } else {
        document.write("탈퇴 취소되었습니다!");
    }
</script>


if else문

else if 문은 두 가지 이상의 조건식과 정해 놓은 조건을 만족하지 않았을때 실행되는 코드로 이루어져 있습니다.

다음은 if else 문의 기본형입니다. 가장 위에 있는 조건식부터 차례대로 검사하면서 만족(true)하는 값이 나오면 그에 해당하는 코드를 실행하고 조건문을 종료합니다.
만족하는 값이 없으면 else 문의 {···} 코드를 실행합니다. 

기본형
{
    const num = 100;
    
    if (num == 90) {
        document.write("실행되었습니다(num==90)");
    } else if (num == 100) {
        document.write("실행되었습니다(num==100)");
    } else if (num == 110) {
        document.write("실행되었습니다(num==110)");
    } else if (num == 120) {
        document.write("실행되었습니다(num==120)");
    } else {
        document.write("실행되었습니다(num==값이없음)");
    }    
}

다음은 질의응답 창을 이용하여 현재는 몇월입니까?를 묻고, if else 문을 사용해 현재 월에 해당하는 계절과 관련된 문구가 출력되도록 작성한 예제입니다.

<script>
    var mon = prompt("현재는 몇 월입니까?.", "");

    if(mon >= 9 && mon <= 11){       //9 ~ 11
        document.write("독서의 계절 가을아네요!!");
    } else if(mon >= 6 && mon <= 8){  //6~8
        document.write("여행가기 좋은 여름이네요!!");
    } else if(mon >= 3 && mon >= 5){  //3~5
        document.write("햇살 가득한 봄이네요!!");
    } esle{                             // 나머지 입력
        document.write("스키의 겨울이네요!!");
    }
</script>
//mon의 값에 따라 문장을 출력합니다.


중첩 if문


조건문 안에 조건문이 있으면 중첩 if 문이라고 합니다. 중첩 if문의 기본형은 다음과 같습니다.

{
    const num = 100;

    if (조건식1) {
        document.write("실행되었습니다.(1)");
        if (조건식2) {
            document.write("실행되었습니다(2)");
            if (조건식3) {
                document.write("실행되었습니다(3)");
            }
        }
    } else {
        document.write("실행되었습니다.(4)");
    }
}

중첩 if 문은 바깥쪽에 있는 조건문인 조건식1을 만족해야만 안쪽에 있는 조건문인 조건문 2를 검사합니다.
만일 안쪽 if 문의 조건식2를 만족하지 않으면 바깥쪽 조건문인 조건식1의 중괄호 안에 있는 코드만 실행하고 종료됩니다.
이를 사용하면 예를들어 로그인 환경에서 방문자로부터 아이디와 비밀번호를 받아서 한번에 처리하지 않아도 됩니다. 아이디가 일치하지않으면 비밀번호 일치여부까지 검사하지 않아도 되기 때문입니다.


다음은 중첩 if문을 이용해 아이디와 비밀번호가 일치했을때 환영문구가 출력하고, 일치하지않을경우 새로 고침이 되도록 작성했습니다.

<script>
    var id = "user123";    //아이디  
    var pw = "123123"       //비밀번호

    var user_id = prompt("아이디는?", "");      //아이디 입력
    var user_pw = prompt("비밀번호는?". "");    //비밀번호 입력
    
    if(id == user_id) {            //아이디가 일치할경우 실행
    if(pw == user_pw) {
        document.write(user_id+"님 환영합니다.");    //비밀번호가 일치할경우 실행            
        } else {
            alert("비밀번호가 틀렸습니다.");           //비밀번호가 일치하지않을경우 실행
            location.reload();             // 브라우저 새로 고침
        }
    } else {
        alert("아이디가 일치하지 않습니다.");       //아이디가 일치하지 않을경우 실행
        location.reload();
    }
</script>
