01. 변수 : 데이터 저장

{
    var x = 100;          //변수 x에 숫자 100을 저장
    var y = 200;          //변수 y에 숫자 200을 저장
    var z = "javascript";   //변수 z에 문자 javascript를 저장

    document.write(** 01. 변수 **);
    document.write(x);
    document.write(y);
    document.write(z);
}

02. 변수 : 데이터 저장 + 데이터 변경

{
    let x = 100;
    let y = 200;
    let z = "javascript"
    x = 300;   // 아래 나중에 입력한 값으로 데이터 변경(변경시에는 let이나 var 쓰지않음 쓸경우 에러남)
    y = 400;
    z = "jquery";

    document.write("*** 02. 변수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
}

03. 변수 : 데이터 저장 + 데이터 변경 + 데이터 추가

{
    let x = 100;
    let y = 200;
    let z = "javascript";

    x += 300; //기존의 저장값에 데이터를 추가 해줄수있음
    y += 400;
    z += "jquery"; //문자의 경우 숫자와 달리 더하는게 불가능하므로 뒤에 문자열이 추가됨

    document.write("*** 03. 변수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
}

04. 변수 : 변수의 종류 = 지역변수 + 전역변수 + 매개변수

{
    let x = 100; //전역변수는 지역변수에 영향을 받지않는다.
    let y = 200; //전역변수   

    function func() {
        let x = 100; //지역변수
        let z = "javascript";
        x = 200; //지역변수 100->200
        y = 300; //전역변수 200->300

        document.write("
함수 안
");
        document.write(x);
        document.write(y);
        document.write(z);
    }
    func();

    document.write("함수 밖
");
    document.write(x);
    document.write(y);
    document.write(z);
}

17. 객체 : 데이터를 저장(키와 값) : 표현 방법8 : 객체 속에 함수

{
    const obj = {
        a: 100,
        b: [200,300],
        c: {x: 400, y:500},
        d: "javascript",
        e: function(){
            document.write("javascript가 실행되었습니다.
");
        },
        f: function(){
            document.write(obj.d + "가 실행되었습니다.
");
        },
        g: function(){
            document.write(this.d + "가 실행되었습니다.
");
        }
        };

    document.write("
*** 17. 객체 ***
");
    document.write(obj.a);
    document.write(obj.b);
    document.write(obj.b[0]);
    document.write(obj.b[1]);
    document.write(obj.c.x);
    document.write(obj.c.y);
    obj.e();
    obj.f();
    obj.g();                        
}
