for문을 사용해서 구구단 신나게 출력하기

for (let i = 2; i <= 9; i++){
    document.write("
");

    for (let j = 1; j <= 9; j++) {
        let sum = i * j;
        document.write(i + " * " + j + " = " + sum, "
")
    }
}

while문을 사용해서 구구단 출력하기

let h = 2
while (h <= 9) {
    document.write("
");
    let g = 1
    while (g <= 9) {
        let sum = h * g;
        document.write(h + "X" + g + "=" + sum, "
")
        g++
    }
    h++
}
